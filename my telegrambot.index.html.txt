    <script>
        const menuBtn = document.getElementById('menu-btn');
        const mobileMenu = document.getElementById('mobile-menu');
        menuBtn.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
        });

        const tooltipTitleCallback = (tooltipItems) => {
            const item = tooltipItems[0];
            if (!item) return '';
            let label = item.chart.data.labels[item.dataIndex];
            return Array.isArray(label) ? label.join(' ') : label;
        };

        const serviceDetails = document.getElementById('service-details').children;
        const servicesChartCanvas = document.getElementById('servicesChart');
        const servicesChart = new Chart(servicesChartCanvas.getContext('2d'), {
            type: 'doughnut',
            data: {
                labels: ['خدمات حقوقی', 'خدمات بازرگانی', 'خدمات عمومی و پشتیبانی'],
                datasets: [{
                    data: [40, 35, 25],
                    backgroundColor: ['#3f51b5', '#7e57c2', '#e57373'],
                    borderColor: '#ffffff',
                    borderWidth: 5,
                    hoverBorderColor: '#f8f9fa',
                    hoverOffset: 15
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                cutout: '65%',
                plugins: {
                    legend: {
                        position: 'bottom',
                        labels: {
                            font: { family: 'Vazirmatn', size: 14 },
                            color: '#495057'
                        }
                    },
                    tooltip: {
                        bodyFont: { family: 'Vazirmatn' },
                        titleFont: { family: 'Vazirmatn' },
                        callbacks: { title: tooltipTitleCallback },
                        backgroundColor: '#343a40',
                        titleColor: '#ffffff',
                        bodyColor: '#ced4da',
                        padding: 10,
                        cornerRadius: 5
                    }
                }
            }
        });

        servicesChartCanvas.onmousemove = (evt) => {
            const points = servicesChart.getElementsAtEventForMode(evt, 'nearest', { intersect: true }, true);
            for (const detail of serviceDetails) {
                 detail.style.transform = 'scale(1)';
                 detail.style.boxShadow = 'none';
            }
            if (points.length) {
                const firstPoint = points[0];
                const label = servicesChart.data.labels[firstPoint.index];
                let serviceKey = '';
                if(label.includes('حقوقی')) serviceKey = 'legal';
                if(label.includes('بازرگانی')) serviceKey = 'commercial';
                if(label.includes('عمومی')) serviceKey = 'support';
                
                const activeDetail = document.querySelector(`[data-service="${serviceKey}"]`);
                if(activeDetail){
                    activeDetail.style.transform = 'scale(1.05)';
                    activeDetail.style.boxShadow = '0 10px 15px -3px rgb(0 0 0 / 0.1)';
                }
            }
        };

        const growthChartCtx = document.getElementById('growthChart').getContext('2d');
        new Chart(growthChartCtx, {
            type: 'line',
            data: {
                labels: ['فاز اولیه', 'سال دوم', 'سال سوم', 'سال چهارم'],
                datasets: [{
                    label: 'رشد پیش‌بینی شده مجموعه',
                    data: [10, 30, 60, 100],
                    fill: true,
                    backgroundColor: 'rgba(63, 81, 181, 0.1)',
                    borderColor: '#3f51b5',
                    tension: 0.4,
                    pointBackgroundColor: '#3f51b5',
                    pointRadius: 5
                }]
            },

        // Gemini Modal Logic
        const modal = document.getElementById('gemini-modal');
        const modalCloseBtn = document.getElementById('gemini-modal-close');
        const modalTitle = document.getElementById('gemini-modal-title');
        const modalPrompt = document.getElementById('gemini-modal-prompt');
        const modalInput = document.getElementById('gemini-modal-input');
        const modalSubmit = document.getElementById('gemini-modal-submit');
        const modalLoader = document.getElementById('gemini-modal-loader');
        const modalOutput = document.getElementById('gemini-modal-output');

        let currentPromptBuilder;

        function openModal(config) {
            modalTitle.textContent = config.title;
            modalPrompt.textContent = config.prompt;
            modalInput.placeholder = config.placeholder;
            currentPromptBuilder = config.promptBuilder;
            modal.classList.remove('hidden');
        }

        function closeModal() {
            modal.classList.add('hidden');
            modalInput.value = '';
            modalOutput.innerHTML = '';
        }

        modalCloseBtn.addEventListener('click', closeModal);
        window.addEventListener('click', (event) => {
            if (event.target === modal) {
                closeModal();
            }
        });

                placeholder: 'مثال: چگونه با بودجه بازاریابی محدود، مشتریان بیشتری جذب کنیم؟',

                نقاط قوت: تیم متخصص و یکپارچه، ساختار چابک، موقعیت استراتژیک در منطقه ویژه.

                فرصت‌ها: رشد تقاضا برای برون‌سپاری، مزایای قانونی منطقه، امکان شبکه‌سازی.

                بر اساس این پروفایل، لطفاً برای چالش یا هدف زیر، چند راهکار عملی و خلاقانه به زبان فارسی ارائه دهید: "${userInput}"`
            });
        });

        modalSubmit.addEventListener('click', async () => {
            const userInput = modalInput.value;
            if (!userInput.trim()) {
                modalOutput.textContent = 'لطفا ورودی را تکمیل کنید.';
                return;
            }

            const prompt = currentPromptBuilder(userInput);
            
            modalLoader.classList.remove('hidden');
            modalOutput.innerHTML = '';
            modalSubmit.disabled = true;

            try {
                const chatHistory = [{ role: "user", parts: [{ text: prompt }] }];
                const payload = { contents: chatHistory };
                const apiKey = ""; 
                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                
                if (!response.ok) {
                    throw new Error(`API request failed with status ${response.status}`);
                }

                const result = await response.json();
                
                if (result.candidates && result.candidates.length > 0) {
                    const text = result.candidates[0].content.parts[0].text;
                    modalOutput.innerHTML = text.replace(/\n/g, '<br>');
                } else {
                    throw new Error('پاسخ معتبری از سرویس دریافت نشد.');
                }

            } catch (error) {
                console.error("Error calling Gemini API:", error);
                modalOutput.textContent = 'متاسفانه در حال حاضر امکان برقراری ارتباط با سرویس هوشمند وجود ندارد. لطفا بعدا تلاش کنید.';
            } finally {
                modalLoader.classList.add('hidden');
                modalSubmit.disabled = false;
            }
        });

    </script>
